P1002 过河卒

题目地址：https://www.luogu.com.cn/problem/P1002

题目截图：


题目分析与题解：


这道题一眼看上去最容易想到的就是用深搜。
首先把马控制的点标记一下，以便遇到马控制的点后递归回溯。

location[x][y]=1;
    location[x+2][y+1]=1;
    location[x+1][y+2]=1;
    if(x+2>=0 && y-1>=0)
        location[x+2][y-1]=1;
    if(x+1>=0 && y-2>=0)
        location[x+1][y-2]=1;
    if(x-2>=0 && y-1>=0)
        location[x-2][y-1]=1;
    if(x-1>=0 && y-2>=0)
        location[x-1][y-2]=1;
    if(x-2>=0 && y+1>=0)
        location[x-2][y+1]=1;
    if(x-1>=0 && y+2>=0)
        location[x-1][y+2]=1;

然后就直接定义dfs函数，在main函数中调用dfs(0,0)，递归回溯就好了。
但是，这道题用深搜不行，有两个测试点会超时。


所以就需要用其他方法。
这道题可以用动态规划（递推）来做。
题目要求的是从 (0,0) 点到 (n,m) 点共有多少条路径，而 (n,m) 点的走法是由到点 (n-1,m) 和 (n,m-1) 的走法决定的。
所以就是，状态转移方程：DP__i,j=DP__i_−1,_j+DP__i,_j_−1

递推公式就是list[i][j]=list[i-1][j]+list[i][j-1].
通过该递推公式可以从 (0,0) 点依次遍历求出所有点的走法，最终求得 (n,m) 点共有多少种走法。
d大致图示就是：

1 1 1 1 1 1 1
1 0 X 1 X 1 2
1 X 0 1 1 X 2
1 1 1 M 1 1 3
1 X 1 1 0 X 3
1 1 X 1 X 0 3
1 2 2 3 3 3 6

代码：

深搜代码：

#include <iostream>
using namespace std;
int location[25][25]={0},sum=0,n,m;

int dfs(int x,int y){
    if(x==n && y==m){
        sum++;
        return 0;
    }
    if((x>n || y>m) || location[x][y]==1)
        return 0;
    dfs(x+1,y);
    dfs(x,y+1);

    return 0;
}

int main() {
    int x,y;
    cin>>n>>m>>x>>y;
    location[x][y]=1;
    location[x+2][y+1]=1;
    location[x+1][y+2]=1;
    if(x+2>=0 && y-1>=0)
        location[x+2][y-1]=1;
    if(x+1>=0 && y-2>=0)
        location[x+1][y-2]=1;
    if(x-2>=0 && y-1>=0)
        location[x-2][y-1]=1;
    if(x-1>=0 && y-2>=0)
        location[x-1][y-2]=1;
    if(x-2>=0 && y+1>=0)
        location[x-2][y+1]=1;
    if(x-1>=0 && y+2>=0)
        location[x-1][y+2]=1;
    dfs(0,0);
    cout<<sum<<endl;

    return 0;
}

DP代码：

#include <iostream>
using namespace std;
int location[25][25]={0};
long long int l[25][25]={0};

int main() {
    int x,y,n,m;
    cin>>n>>m>>x>>y;
    location[x][y]=1;
    location[x+2][y+1]=1;
    location[x+1][y+2]=1;
    if(x+2>=0 && y-1>=0)
        location[x+2][y-1]=1;
    if(x+1>=0 && y-2>=0)
        location[x+1][y-2]=1;
    if(x-2>=0 && y-1>=0)
        location[x-2][y-1]=1;
    if(x-1>=0 && y-2>=0)
        location[x-1][y-2]=1;
    if(x-2>=0 && y+1>=0)
        location[x-2][y+1]=1;
    if(x-1>=0 && y+2>=0)
        location[x-1][y+2]=1;
    for(int i=0;i<=n;i++){
        for(int j=0;j<=m;j++){
            if(i==0 && j==0){
                l[i][j]=1;
                continue;
            }
            if(location[i][j]==1)
                continue;
            if(i==0){
                l[i][j]=l[0][j-1];
                continue;
            }
            if(j==0){
                l[i][j]=l[i-1][0];
                continue;
            }
            l[i][j]=l[i-1][j]+l[i][j-1];
        }
    }

    cout<<l[n][m]<<endl;


    return 0;
}
